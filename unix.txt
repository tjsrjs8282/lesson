cd #root, 최상위 디렉토리로 이동
cd .. #상위 디렉토리로 이동
cd . #하위 디렉토리로 이동
cd test_folder #상대경로 이동법, 현재 폴더에 존재하는 test_folder로 이동

ls #현재 디렉토리 내에 있는 파일과 디렉토리의 리스트를 보여준다.

ls -l # 숨겨진 파일까지 드러낸다.
ls -a # 파일의 자세한 정보들을 출력(링크 수, 소유자, 권한, 모드, 최종 수정시간)
ls -al # a와 l을 합한 옵션.

pwd # 현재 위치한 경로 확인

touch test.txt # text.txt 파일 생성
